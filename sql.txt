DELETE:it is used to delete record from table
syntax:
    delete from <table name> where <condition>
    eg:
        delete record from employee table with empid1001
        delete from employee where empid=1001;

UPDATE:used to modify record of table
syntax:
    update <table_name> set <column_name>=<value>, <column_name>=<value> whre <condition>;
    e.g
        update employee set salary=60000 where empid=1002;


TRUNCATE:used to delete all record from table but structure remains
syntax:
    truncate table<table_name>;
    e.g
        truncate table employee;

DROP:it is used to delete database object
syntax:
    drop table<table_name>
    eg:
        drop table employee;


ALTER:it is used to modify structure of database object

create table login
(
    userid varchar(50) primary key,
    password varchar(50)
);

1.use of alter command to add a new column
syntax:
    alter table <table_name> add <column_name> <data_type>
    eg:
        alter table login add usertype varchar(50);

2. use of alter command to drop a column
syntax:
    alter table<table_name> drop column <column_name>
    eg  
        alter table login drop column usertype;

3.use of alter command to modify column
syntax:
    alter table <table_name> modify <column_name> <new_data_type>;
    eg:
        alter table login modify password varchar(30);

4.use of alter command to rename the column
syntax:
    alter table <table_name> change <column_old_name> <column_new_name> <data_type>;
    eg:
        alter table  login change password pwd varchar(30);

5.use of alter command to add new column after a given column
syntax:
    alter table <table_name> add <new_column_name> <data_type> after <given_column_name>;
    eg:
        alter table login add usertype varchar(50) after userid;

Natural join: when we perform this operation common records from both are display


task: we want to select empname from staff table and pname from product table on the basis of empid

select staff.empname, product.pname from staff, product where staff.empid=product.empid;

LEFT JOIN:
all records pf left table are display and matching records of right table are display

select staff.empname, product.pname from staff left join product on staff.empid=product.empid; 


Right join: all records pf right table are display and matching records of left table are display
select staff.empname, product.pname from staff right join product on staff.empid=product.empid; 


Full join: all records of both table are display. it is like as union of left join and right join
select staff.empname, product.pname from staff left join product on staff.empid=product.empid 
union
select staff.empname, product.pname from staff right join product on staff.empid=product.empid;
